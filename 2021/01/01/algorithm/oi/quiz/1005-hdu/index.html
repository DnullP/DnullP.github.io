求一个递推公式得第n项，数据范围不能使用线性算法，所以采用矩阵快速幂的做法

虽然是很久以前就会的算法，但是实际写起来还是踩了不少坑，此外在写对拍程序时遇到了c++输出的缓冲问题，以至于数据生成器一直无法生成数据，非常消耗人的心态

另外此题貌似可以通过找规律来解决，但是我并不想花时间去玩找规律的游戏

本题踩的一些坑

  • 重载矩阵乘法时，下标搞错
  • 需要特殊判断n=1的情况

代码如下：

#include <cstdio>
#include <iostream>
#include <map>
#include <cstring>

using namespace std;

struct rec_2
{
    int a_11 = 0, a_12 = 0, a_21 = 0, a_22 = 0;

    rec_2 operator*(rec_2 b)
    {

        rec_2 c(0, 0, 0, 0);

        c.a_11 = (a_11 * b.a_11 + a_12 * b.a_21) % 7;
        c.a_12 = (a_11 * b.a_12 + a_12 * b.a_22) % 7;
        c.a_21 = (a_21 * b.a_11 + a_22 * b.a_21) % 7;
        c.a_22 = (a_21 * b.a_12 + a_22 * b.a_22) % 7;

        return c;
    }
    rec_2(int a, int b, int c, int d)
    {
        a_11 = a;
        a_21 = b;
        a_12 = c;
        a_22 = d;
    }
};

rec_2 qpow(rec_2 x, long long a)
{
    rec_2 base = x;
    rec_2 ans(1, 0, 0, 1);

    while (a)
    {
        if (a & 1)
            ans = ans * base;
        base = base * base;
        a >>= 1;
    }
    return ans;
}

int main()
{
    /* FILE *stream1;
    freopen_s(&stream1, "data.in", "r", stdin);
    freopen_s(&stream1, "a.out", "w", stdout); */

    while (true)
    {
        int A, B;
        long long n;

        cin >> A >> B >> n;

        if (A == 0 && B == 0 && n == 0)
            break;

        if(n==1){
            cout<<1<<endl;
            continue;
        }
        if(n==0){
            continue;
        }
        rec_2 base(A, B, 1, 0);
        rec_2 f(1, 0, 1, 0);

        f = f * qpow(base, n - 2);

        cout << f.a_11 % 7 << endl;
    }

    //fclose(stdin);
    //fclose(stdout);
}
