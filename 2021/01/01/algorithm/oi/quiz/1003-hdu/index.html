照理说应当是一道相当简单入门的线性题，但是实际做的时候还是踩了不少坑

几种没有考虑到的特殊情况：

  • 数列全为负数的情况下，应当找到最小负数
  • 线性过程中的t_sum应该在首项赋值

代码如下：

#include <cstdio>
#include <iostream>
#include <algorithm>
#include <vector>
#include <cstring>
#include <string>
#include <string.h>
#include <ctime>
#include <cmath>
#include <stdlib.h>
#include <iomanip>

using namespace std;

struct ANS_
{
    int begin = 0;
    int end = 0;
    int sum = 0;
};

ANS_ caculate(int seq[], int l)
{
    ANS_ ans;

    int t_sum = seq[0];
    ans.sum = seq[0];
    int last = 0;
    for (int i = 0; i < l; i++)
    {
        if (last != i)
            t_sum += seq[i];

        //cout << t_sum << endl;

        if (t_sum >= 0 && t_sum > ans.sum)
        {
            ans.sum = t_sum;
            ans.end = i;
            ans.begin = last;
        }
        else if (t_sum < 0)
        {
            if (t_sum > ans.sum)
            {
                ans.sum = t_sum;
                ans.end = i;
                ans.begin = last;
            }

            t_sum = seq[i + 1];
            last = i + 1;
        }
    }
    return ans;
}

int main()
{
    //FILE *stream1;
    //freopen_s(&stream1, "data.in", "r", stdin);
    //freopen_s(&stream1, "a.out", "w", stdout);

    int n;
    int seq[100005];

    cin >> n;

    for (int j = 0; j < n; j++)
    {
        int l;
        cin >> l;

        for (int i = 0; i < l; i++)
        {
            cin >> seq[i];
        }
        ANS_ ans = caculate(seq, l);
        cout << "Case " << j + 1 << ':' << endl
             << ans.sum << ' ' << ans.begin + 1 << ' ' << ans.end + 1 << endl;

        if (j != n - 1)
            cout << endl;
        // << endl;
    }
    fclose(stdin);
    fclose(stdout);
}
