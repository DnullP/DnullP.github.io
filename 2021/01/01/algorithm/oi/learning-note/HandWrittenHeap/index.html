手写堆模板

手写了一个堆的类模板，以后遇到需要用堆的情况会直接使用

#include <list>

using namespace std;

template <typename T>
class heap
{
private:
    bool SH;
    int size_;
    T *node_ = new T[1000005];

    void update(int node);
    void update_down(int node);

public:
    heap(bool);
    heap(bool, list<T>);

    void insert(T);
    void extract();
    T GetTop();
    void remove(int);
    void print_all();

    ~heap();
};

template <typename T>
heap<T>::heap(bool BigOrSmall, list<T> init_)
{
    this->SH = BigOrSmall;
}

template <typename T>
heap<T>::heap(bool BigOrSmall)
{
    this->SH = BigOrSmall;
}

template <typename T>
heap<T>::~heap() {}

template <typename T>
void heap<T>::update(int node)
{
    if (node == 1)
        return;

    int father = (node ^ 1) / 2;
    if ((*(node_ + father) > *(node_ + node)) == SH)
    {
        swap(*(node_ + father), *(node_ + node));

        update(father);
    }
}

template <typename T>
void heap<T>::update_down(int node)
{
    int Lc = 2 * node;
    int Rc = 2 * node + 1;

    int target = node;

    if (Rc <= size_)
    {
        if ((*(node_ + target) > *(node_ + Rc)) == SH)
        {
            target = Rc;
        }
    }
    if (Lc <= size_)
    {
        if ((*(node_ + target) > *(node_ + Lc)) == SH)
        {
            target = Lc;
        }
    }
    if (target == node)
        return;
    swap(*(node_ + node), *(node_ + target));
    update_down(target);
}

template <typename T>
void heap<T>::insert(T val)
{

    *(node_ + (++size_)) = val;
    update(size_);
}

template <typename T>
T heap<T>::GetTop()
{
    return *(node_ + 1);
}

template <typename T>
void heap<T>::extract()
{
    swap(*(node_ + 1), *(node_ + size_));

    size_--;
    update_down(1);
}

template <typename T>
void heap<T>::remove(int node)
{
    swap(*(node_ + node), *(node_ + size_));

    size_--;
    update_down(node);
    update(node);
}

template <typename T>
void heap<T>::print_all()
{
    for (int i = 0; i < size_; i++)
    {
        cout << *(node_ + i + 1) << ' ';
    }
    cout << endl;
}

int main(){}
