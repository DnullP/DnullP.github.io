ST表

由倍增算法和思想衍生出的一种适用于区间查询的基本数据类型

在倍增的思想中，通过成倍的增长来加快某一种过程，而在st表中，通过成倍的增加长度来扩大一个区间的覆盖长度，获得一个更大区间的某一特征。

以区间最大值为例子
现在我用一个二维数组f[i][j]f[i][j]f[i][j]表示以数列的第i项开头的，长度为2j2^j2j的一个区间中的最大值，然后我们可以得到在[l,r][l,r][l,r]的范围内的最大的2的幂次方为log2(r−l+1)log_2(r-l+1)log2​(r−l+1)，同时可以保证这个数字大于l+r2\frac{l+r}{2}2l+r​。
所以我们可以用
f[l][log2(r−l+1)/2]f[l][log_2(r-l+1)/2]f[l][log2​(r−l+1)/2]
和
f[r−2log2(r−l+1)/2][log2(r−l+1)/2]f[r-2^{log_2(r-l+1)/2}][log_2(r-l+1)/2]f[r−2log2​(r−l+1)/2][log2​(r−l+1)/2]

来表示区间[l,r][l,r][l,r]，这两个区间的合集就是[l,r][l,r][l,r]，而两者中间的重叠部分并不影响总区间的最大值，所以可以用O(1)O(1)O(1)的时间复杂度对区间最值进行查询。

以下是洛谷ST表模板题的代码：

#include <iostream>
#include <cmath>

using namespace std;

int main()
{
    int f[100005][21];
    int n, m;
    cin >> n >> m;
    for (int i = 1; i <= n; i++)
    {
        cin >> f[i][0];
    }
    for (int i = 1; i <= 21; i++)
    {
        for (int j = 1; j + (1 << i) - 1 <= n; j++)
        {
            f[j][i] = f[j][i - 1];
            f[j][i] = max(f[j][i], f[j + (1 << i) / 2][i - 1]);
        }
    }
    for (int i = 1; i <= m; i++)
    {
        int l, r;
        cin >> l >> r;
        int t = log(r - l + 1) / log(2);
        cout << max(f[l][t], f[r - (1 << t) + 1][t]) << endl;
    }
}
